Exercici 9
  • 9- Escriviu un programa que demani una frase i mostri quantes vegades apareix la lletra a.


function ex0109() {
    frase = prompt("Escriu una frase:",);
    comptador = 0; i = 0; // inicialitzem variables a 0

    // recorrem la frase, caracter a caracter
    while (i <= frase.length) {
        if (frase.substr(i, 1) == 'a' || frase.substr(i, 1) == 'à' || frase.substr(i, 1) == 'A' || frase.substr(i, 1) == 'À') {
            comptador++;
        }
        i++;
    }
    // mostrem per pantalla el resultat
    alert('La frase conté ' + comptador + ' a.');
    document.write('La frase conté ' + comptador + ' a.');
}

Exercici 15
  • 15- Escriviu un programa que mostri per pantalla els divisors d'un número donat.


function ex0115() {
    numero = parseInt(prompt("Introdueix el número a analitzar"));
    i = 1; divisor = "";
    while (i < numero) {
        if (numero % i == 0) {
            divisor = divisor + i.toString() + ",";
        }
        i++;
    }
    document.write("El teu número és divisible entre: " + divisor + numero + ".");
}
Exercici 16
  • 16- Escriviu un programa que mostri per pantalla els divisors comuns de dos números donats.


function ex0116() {
    numero1 = parseInt(prompt("Introdueix el primer número"));
    numero2 = parseInt(prompt("Introdueix el segon número"));
    i = 0; divisor = "";
    if (numero1 > numero2) {
        while (i <= numero2) {
            if (numero1 % i == 0 && numero2 % i == 0) {
                divisor = divisor + i.toString() + ", ";
            }
            i++;
        }
    }
    else {
        while (i <= numero1) {
            if (numero2 % i == 0 && numero1 % i == 0) {
                divisor = divisor + i.toString() + ", ";
            }
            i++;
        }
    }
    document.write("Els dos números són divisibles entre: " + divisor);
}
